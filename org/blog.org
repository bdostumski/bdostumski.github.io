(require 'subr-x)
(require 'seq)

(defun blog--posts-dir ()
  "Return absolute path to the posts directory (org/posts)."
  (let* ((this-file (or (buffer-file-name)
                        (expand-file-name "blog.org" default-directory)))
         (org-dir  (file-name-directory this-file)))
    (expand-file-name "posts" org-dir)))

(defun blog--read-org-metadata (file)
  "Return (TITLE DATE SLUG TAGS) for FILE.
TAGS is a list of strings parsed from #+TAGS: header."
  (with-temp-buffer
    (insert-file-contents file)
    (goto-char (point-min))
    (let* ((title (or (when (re-search-forward "^#\\+TITLE:[ \t]+\\(.+\\)$" nil t)
                        (string-trim (match-string 1)))
                      (file-name-base file)))
           (_     (goto-char (point-min)))
           (date  (when (re-search-forward "^#\\+DATE:[ \t]+\\(.+\\)$" nil t)
                    (string-trim (match-string 1))))
           (_     (goto-char (point-min)))
           (tags  (when (re-search-forward "^#\\+TAGS:[ \t]+\\(.+\\)$" nil t)
                    (split-string (string-trim (match-string 1)) "[ \t,]+" t)))
           (slug  (file-name-base file)))
      (list title date slug (or tags '("untagged"))))))

(defun blog-generate-tags ()
  "Generate an Org page grouped by tags with post counts."
  (let* ((base-dir  (file-name-as-directory (blog--posts-dir)))
         (files     (directory-files base-dir t "\\.org\\'" t))
         (all-posts (mapcar #'blog--read-org-metadata files))
         (tag-table (make-hash-table :test 'equal)))

    ;; Group posts by tag
    (dolist (post all-posts)
      (let ((title (nth 0 post))
            (date  (nth 1 post))
            (slug  (nth 2 post))
            (tags  (nth 3 post)))
        (dolist (tag tags)
          (puthash tag
                   (cons (list title date slug) (gethash tag tag-table))
                   tag-table))))

    ;; Sort tags alphabetically, build output
    (let ((sorted-tags (sort (hash-table-keys tag-table) #'string<)))
      (mapconcat
       (lambda (tag)
         (let* ((posts (gethash tag tag-table))
                (count (length posts))
                (sorted-posts
                 (sort posts
                       (lambda (a b)
                         (let ((da (nth 1 a))
                               (db (nth 1 b)))
                           (cond
                            ((and da db) (string< da db))
                            (da t)
                            (t nil)))))))
           (concat
            (format "** %s (%d)\n" tag count)
            (mapconcat
             (lambda (post)
               (let ((title (nth 0 post))
                     (date  (nth 1 post))
                     (slug  (nth 2 post)))
                 (format "- %s[[./posts/%s.html][%s]]"
                         (if date (format "%s – " date) "")
                         slug
                         title)))
             sorted-posts
             "\n"))))
       sorted-tags
       "\n\n"))))

;; Return the generated content to Org Babel
(blog-generate-tags)
